#include <bits/stdc++.h>
using namespace std;
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n,x,y;cin>>n>>x>>y;
    long long ans=0;
    while(n--){
        int num;cin>>num;
        if(num>= x && num <=y)
            ans+=x;
    }
    cout<<ans<<"\n";

    return 0;
}
